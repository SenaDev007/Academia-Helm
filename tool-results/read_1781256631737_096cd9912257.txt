   488→/**
   489→ * État d'un événement dans l'Outbox
   490→ */
   491→export type OutboxEventStatus = 
   492→  | 'PENDING'      // En attente d'envoi
   493→  | 'SENT'         // Envoyé au serveur
   494→  | 'ACKNOWLEDGED' // Confirmé par le serveur
   495→  | 'FAILED'       // Échec d'envoi
   496→  | 'CONFLICT';    // Conflit détecté

   498→/**
   499→ * Événement Outbox (Desktop → Server)
   500→ * 
   501→ * Pattern Outbox : chaque modification locale est enregistrée
   502→ * dans une table outbox avant d'être synchronisée
   503→ */
   504→export interface OutboxEvent {
   505→  id: string; // UUID généré côté Desktop
   506→  tenantId: string;
   507→  entityType: SyncEntityType;
   508→  entityId: string; // ID de l'entité modifiée
   509→  operation: SyncOperationType;
   510→  /**
   511→   * Données de l'entité (JSON)
   512→   * Pour CREATE/UPDATE : données complètes
   513→   * Pour DELETE : seulement l'ID
   514→   */
   515→  payload: Record<string, any>;
   516→  /**
   517→   * Version locale (timestamp ou version number)
   518→   * Utilisé pour la détection de conflits
   519→   */
   520→  localVersion: number;
   521→  /**
   522→   * Timestamp de création côté Desktop
   523→   */
   524→  createdAt: string;
   525→  /**
   526→   * Timestamp de dernière tentative d'envoi
   527→   */
   528→  lastAttemptAt?: string;
   529→  /**
   530→   * Nombre de tentatives d'envoi
   531→   */
   532→  attemptCount: number;
   533→  status: OutboxEventStatus;
   534→  /**
   535→   * Message d'erreur si échec
   536→   */
   537→  errorMessage?: string;
   538→}

   541→ * Requête de synchronisation montante (Desktop → Server)

   543→export interface SyncUpRequest {
   544→  /**
   545→   * ID du client Desktop (identifiant unique de l'installation)
   546→   */
   547→  clientId: string;
   548→  /**
   549→   * Liste des événements à synchroniser
   550→   */
   551→  events: OutboxEvent[];
   552→  /**
   553→   * Timestamp de la dernière synchronisation réussie
   554→   * Permet au serveur de savoir quels changements envoyer en retour
   555→   */
   556→  lastSyncTimestamp?: string;
   557→}
   558→
   559→/**
   560→ * Réponse de synchronisation montante
   561→ */
   562→export interface SyncUpResponse {
   563→  /**
   564→   * Événements acceptés (avec leur ID serveur)
   565→   */
   566→  acknowledged: Array<{
   567→    outboxEventId: string; // ID original côté Desktop
   568→    serverEventId: string; // ID généré côté serveur
   569→    entityId: string;      // ID final de l'entité (peut différer si création)
   570→    serverVersion: number; // Version serveur
   571→  }>;
   572→  /**
   573→   * Événements en conflit
   574→   */
   575→  conflicts: Array<{
   576→    outboxEventId: string;
   577→    reason: string;
   578→    serverData?: Record<string, any>; // Données serveur pour résolution
   579→  }>;
   580→  /**
   581→   * Événements rejetés (erreur de validation)
   582→   */
   583→  rejected: Array<{
   584→    outboxEventId: string;
   585→    reason: string;
   586→  }>;
   587→  /**
   588→   * Timestamp de synchronisation (pour la prochaine sync)
   589→   */
   590→  syncTimestamp: string;
   591→}
   592→

   594→ * Requête de synchronisation descendante (Server → Desktop)

   596→export interface SyncDownRequest {
